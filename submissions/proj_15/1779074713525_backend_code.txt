```java
package com.raincheck.finalyearproj.controller;

import com.baomidou.mybatisplus.core.conditions.query.LambdaQueryWrapper;
import com.raincheck.finalyearproj.common.Result;
import com.raincheck.finalyearproj.common.UserContext;
import com.raincheck.finalyearproj.model.entity.ChatMessage;
import com.raincheck.finalyearproj.service.ChatMessageService;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.web.bind.annotation.*;

import java.util.List;

@RestController
@RequestMapping("/api/chat")
public class ChatController {

    @Autowired
    private ChatMessageService chatMessageService;

    // 获取我与某个人的全部聊天记录
    @GetMapping("/history/{contactId}")
    public Result<List<ChatMessage>> getChatHistory(@PathVariable Integer contactId) {
        Integer myId = UserContext.getUserId();

        // 查询条件：(sender = me AND receiver = contact) OR (sender = contact AND receiver = me)
        List<ChatMessage> history = chatMessageService.list(new LambdaQueryWrapper<ChatMessage>()
                .and(wrapper -> wrapper
                        .eq(ChatMessage::getSenderId, myId)
                        .eq(ChatMessage::getReceiverId, contactId)
                )
                .or(wrapper -> wrapper
                        .eq(ChatMessage::getSenderId, contactId)
                        .eq(ChatMessage::getReceiverId, myId)
                )
                .orderByAsc(ChatMessage::getCreatedAt) // 按时间正序，旧的在上面
        );

        return Result.success(history);
    }
}
```

```java
@RestController
@RequestMapping("/api/student/projects")
public class StudentProjectController {

    @Autowired
    private VersionLogService versionLogService; // 注入日志服务

    @Autowired
    private UserService userService;

    // A. 获取“我的项目”详情 (已有的)
    @GetMapping("/my")
    public Result<ProjectVO> getMyProject() {
        Integer studentId = UserContext.getUserId();

        // 使用你自定义的 Service 方法进行联表查询
        // 如果还没写联表，可以先用这种简单方式凑效：
        Project project = projectService.getProjectVoById(studentId);

        if (project == null) return Result.error(404, "No project found");

        ProjectVO vo = new ProjectVO();
        BeanUtils.copyProperties(project, vo);

        // 假设你有 StaffService，根据 staffId 查名字
        if (project.getSupervisorId() != null) {
            User staff = userService.getById(project.getSupervisorId());
            vo.setSupervisorName(staff != null ? staff.getName() : "Not Assigned");
        }

        return Result.success(vo);
    }

    // B. 学生提交作业 (已有的)
    @PostMapping("/{projectId}/upload")
    public Result<UploadedFile> upload(@PathVariable Integer projectId,
                                       @RequestPart MultipartFile file,
                                       @RequestParam(required = false) String description) {
        Project project = projectService.getById(projectId);

        // 修复点：确保类型一致。建议统一使用 Long 或在比较时转为 String
        if (project == null || !String.valueOf(project.getStudentId()).equals(String.valueOf(UserContext.getUserId()))) {
            return Result.error(403, "Access Denied. Project ID: " + projectId + " does not belong to user: " + UserContext.getUserId());
        }

        // 如果这里报错，说明 uploadSubmission 内部逻辑有问题
        return Result.success(projectService.uploadSubmission(projectId, file, description));
    }

    /**
     * C. 满足需求 3.3 (2): 查看自己的版本记录/进展日志
     * 学生需要看自己什么时候交了什么，以及对应的 GitHub 链接
     */
    @GetMapping("/{projectId}/progress")
    public Result<List<VersionLog>> getMyProgress(@PathVariable Integer projectId) {
        // 安全校验：防止学生 A 查学生 B 的进度
        Project project = projectService.getById(projectId);
        if (project == null || !project.getStudentId().equals(UserContext.getUserId())) {
            return Result.error(403, "Access Denied.");
        }

        // 查询该项目下所有的日志，按时间倒序排列
        List<VersionLog> logs = versionLogService.list(new LambdaQueryWrapper<VersionLog>()
                .eq(VersionLog::getProjectId, projectId)
                .orderByDesc(VersionLog::getCreatedAt));

    }
```

```java
package com.raincheck.finalyearproj.interceptor;

import com.raincheck.finalyearproj.common.UserContext;
import com.raincheck.finalyearproj.utils.JwtUtils;
import io.jsonwebtoken.Claims;
import lombok.extern.slf4j.Slf4j;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Component;
import org.springframework.web.servlet.HandlerInterceptor;

import javax.servlet.http.HttpServletRequest;
import javax.servlet.http.HttpServletResponse;
import java.io.IOException;

@Slf4j
@Component
public class JwtInterceptor implements HandlerInterceptor {

    @Autowired
    private JwtUtils jwtUtils;

    @Override
    public boolean preHandle(HttpServletRequest request, HttpServletResponse response, Object handler) throws IOException {
        if ("OPTIONS".equalsIgnoreCase(request.getMethod())) {
            return true;
        }

        String token = request.getHeader("Authorization");

        if (token != null && token.startsWith("Bearer ")) {
            token = token.substring(7);

            // 解析即验证
            Claims claims = jwtUtils.getClaimsByToken(token);

            if (claims != null) {
                // 验证通过，封装用户信息
                UserContext.UserDTO userDTO = UserContext.UserDTO.builder()
                        .userId((Integer) claims.get("userId"))
                        .email(claims.getSubject())
                        .role((String) claims.get("role"))
                        .build();

                UserContext.setUser(userDTO);
                return true;
            }
        }

        // 验证失败：Token 缺失、过期、签名错误都会走到这里
        response.setContentType("application/json;charset=UTF-8");
        response.setStatus(401);
        response.getWriter().write("{\"code\": 401, \"msg\": \"Unauthorized: Invalid or expired token\"}");
        return false;
    }

    @Override
    public void afterCompletion(HttpServletRequest request, HttpServletResponse response, Object handler, Exception ex) {
        // 关键：请求结束后必须清理，因为 Tomcat 线程池会复用线程
        UserContext.removeUser();
    }
}
```

```java
    @Autowired

    @Value("${github.token}")
    private String token;
    @Value("${github.owner}")
    private String owner;
    @Value("${github.repo}")
    private String repo;

    private final String GITHUB_API_URL = "https://api.github.com/repos/%s/%s/contents/%s";

    /**
     * 1. 创建项目 (学生端)
     * 逻辑：一个学生只能拥有一个处于激活状态的项目。
     */
    @Override
    public void createProject(Project project) {
        // 校验是否已有项目
        Long count = this.count(new LambdaQueryWrapper<Project>()
                .eq(Project::getStudentId, project.getStudentId()));
        if (count > 0) {
            throw new RuntimeException("Error: You already have an active project.");
        }

        project.setCreatedAt(LocalDateTime.now());
        project.setUpdatedAt(LocalDateTime.now());
        this.save(project);
    }

    /**
     * 2. 上传提交物至 GitHub (核心业务)
     * 逻辑：文件推送至 GitHub -> 数据库记录元数据 -> 自动生成版本日志
     */
    @Transactional(rollbackFor = Exception.class)
    @Override
    public UploadedFile uploadSubmission(Integer projectId, MultipartFile file, String description) {
        try {
            // A. GitHub 存储逻辑
            String originalFilename = file.getOriginalFilename();
            String path = "submissions/proj_" + projectId + "/" + System.currentTimeMillis() + "_" + originalFilename;
            String base64Content = Base64.getEncoder().encodeToString(file.getBytes());

            String url = String.format(GITHUB_API_URL, owner, repo, path);

            Map<String, Object> body = new HashMap<>();
            body.put("message", "Student upload: " + originalFilename);
            body.put("content", base64Content);

            HttpHeaders headers = new HttpHeaders();
            headers.set("Authorization", "Bearer " + token);
            headers.set("Accept", "application/vnd.github+json");

            ResponseEntity<Map> response = restTemplate.exchange(url, HttpMethod.PUT, new HttpEntity<>(body, headers), Map.class);

            if (response.getStatusCode() != HttpStatus.CREATED) {
                throw new RuntimeException("GitHub upload failed with status: " + response.getStatusCode());
            }

            // B. 数据库记录：文件表
            Map contentMap = (Map) response.getBody().get("content");
            String githubUrl = (String) contentMap.get("html_url");

            UploadedFile fileRecord = new UploadedFile();
            fileRecord.setProjectId(projectId);
            fileRecord.setFileName(originalFilename);
            fileRecord.setFileUrl(githubUrl);
            fileRecord.setSizeMb((float) file.getSize() / (1024 * 1024));
            fileRecord.setUploadedAt(LocalDateTime.now());
            fileMapper.insert(fileRecord);

            // C. 数据库记录：自动生成的版本日志 (Version Log)
            VersionLog logEntry = new VersionLog();
            logEntry.setProjectId(projectId);
            logEntry.setLogType("UPLOAD");
            logEntry.setCommitLink(githubUrl);
            logEntry.setDescription(description);
            logEntry.setCreatedAt(LocalDateTime.now());
            logEntry.setTimestamp(LocalDateTime.now());
            logMapper.insert(logEntry);

            return fileRecord;

        } catch (Exception e) {
            // 触发 @Transactional 回滚，保证 DB 不会产生孤立记录
            throw new RuntimeException("Upload Process Failed: " + e.getMessage());
        }
    }

    public ProjectVO getProjectVoById(Integer studentId) {
        Project project = this.getOne(new LambdaQueryWrapper<Project>()
                .eq(Project::getStudentId, studentId));;
        if (project == null) return null;

        ProjectVO vo = new ProjectVO();
        BeanUtils.copyProperties(project, vo);

        if (project.getSupervisorId() != null) {
            User supervisor = userMapper.selectById(project.getSupervisorId());
            if (supervisor != null) {
                vo.setSupervisorName(supervisor.getName());
                vo.setSupervisorEmail(supervisor.getEmail()); // 注入导师邮箱
            }
        }

        if (project.getStudentId() != null) {
            User student = userMapper.selectById(project.getStudentId());
            if (student != null) {
                vo.setStudentName(student.getName());
                vo.setStudentEmail(student.getEmail()); // 注入学生邮箱
            }
        }

        return vo;
    }

    public List<ProjectVO> convertToVoList(List<Project> projects) {

                vo.setStudentEmail(student.getEmail());
            }

        }).collect(Collectors.toList());
    }
```

```java
import io.jsonwebtoken.Claims;

import org.springframework.stereotype.Component;

import java.util.Date;
import java.util.HashMap;
import java.util.Map;

@Slf4j
@Component
public class JwtUtils {

    @Value("${jwt.secret:MySecretKey123}") // 建议在yml中定义
    private String secret;

    @Value("${jwt.expiration:86400000}") // 默认24小时
    private long expire;

    // 生成 Token
    public String createToken(Integer userId, String email, String role) {
        Date now = new Date();
        Date expiration = new Date(now.getTime() + expire * 1000);

        Map<String, Object> claims = new HashMap<>();
        claims.put("userId", userId);
        claims.put("role", role);

        return Jwts.builder()
                .setHeaderParam("typ", "JWT")
                .setClaims(claims)
                .setSubject(email)
                .setIssuedAt(now)
                .setExpiration(expiration)
                .signWith(SignatureAlgorithm.HS256, secret)
                .compact();
    }

    /**
     * 校验并解析 Token
     */
    public boolean validateToken(String token) {
        try {
            Jwts.parser().setSigningKey(secret).parseClaimsJws(token);
            return true;
        } catch (Exception e) {
            // 这里会捕获：Token过期、签名错误、格式错误等异常
            return false;
        }
    }

    // JwtUtils.java
    public Claims getClaimsByToken(String token) {
        try {
            return Jwts.parser()
                    .setSigningKey(secret)
                    .parseClaimsJws(token)
                    .getBody();
        } catch (Exception e) {
            // 签名不对、过期、格式不对都会走到这里
            log.error("JWT validation failed: {}", e.getMessage());
            return null;
        }
    }

    /**
     * 获取完整的 Claims
     */
    public Claims getClaims(String token) {
        return Jwts.parser()
                .setSigningKey(secret)
                .parseClaimsJws(token)
                .getBody();
    }
```

```java
package com.raincheck.finalyearproj.websocket;

import com.fasterxml.jackson.databind.JsonNode;
import com.fasterxml.jackson.databind.ObjectMapper;
import com.raincheck.finalyearproj.model.entity.ChatMessage;
import com.raincheck.finalyearproj.service.ChatMessageService;
import lombok.extern.slf4j.Slf4j;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Component;
import org.springframework.web.socket.*;
import org.springframework.web.socket.handler.TextWebSocketHandler;

import java.time.LocalDateTime;
import java.util.concurrent.ConcurrentHashMap;

@Slf4j
@Component
public class ChatWebSocketHandler extends TextWebSocketHandler {

    @Autowired
    private ChatMessageService chatMessageService;

    @Autowired
    private ObjectMapper objectMapper; // Spring 自带的 JSON 处理工具

    // 存储所有在线用户的 WebSocketSession (Key 为 userId)
    private static final ConcurrentHashMap<Integer, WebSocketSession> SESSIONS = new ConcurrentHashMap<>();

    @Override
    public void afterConnectionEstablished(WebSocketSession session) throws Exception {
        Integer userId = (Integer) session.getAttributes().get("userId");
        SESSIONS.put(userId, session);
        log.info("User {} connected to chat.", userId);
    }

    @Override
    protected void handleTextMessage(WebSocketSession session, TextMessage message) throws Exception {
        Integer senderId = (Integer) session.getAttributes().get("userId");
        String payload = message.getPayload();

        // 解析前端发来的 JSON，例如：{"receiverId": 104, "content": "老师好！"}
        JsonNode jsonNode = objectMapper.readTree(payload);
        Integer receiverId = jsonNode.get("receiverId").asInt();
        String content = jsonNode.get("content").asText();

        // 1. 保存消息到数据库
        ChatMessage chatMsg = new ChatMessage();
        chatMsg.setSenderId(senderId);
        chatMsg.setReceiverId(receiverId);
        chatMsg.setContent(content);
        chatMsg.setCreatedAt(LocalDateTime.now());
        chatMsg.setIsRead(false);
        chatMessageService.save(chatMsg);

        // 2. 尝试发给接收者（如果对方在线）
        WebSocketSession receiverSession = SESSIONS.get(receiverId);
        if (receiverSession != null && receiverSession.isOpen()) {
            // 把完整的消息对象转成 JSON 发给对方前端
            String msgJson = objectMapper.writeValueAsString(chatMsg);
            receiverSession.sendMessage(new TextMessage(msgJson));
        }

        // 3. （可选）把消息回传给发送者自己，以确认发送成功
        session.sendMessage(new TextMessage(objectMapper.writeValueAsString(chatMsg)));
    }

    @Override
    public void afterConnectionClosed(WebSocketSession session, CloseStatus status) throws Exception {
        Integer userId = (Integer) session.getAttributes().get("userId");
        if (userId != null) {
            SESSIONS.remove(userId);
            log.info("User {} disconnected from chat.", userId);
        }
    }
}
```
